/* ===== 🌍 全局 i18n 系统 ===== */

let LANG = localStorage.getItem("lang") || "en"
let dict = {}

async function loadLang(lang = LANG){

LANG = lang
localStorage.setItem("lang", LANG)

const res = await fetch("/static/lang/" + LANG + ".json")
dict = await res.json()

applyTranslations()
}

function applyTranslations(){

document.querySelectorAll("[data-i18n]").forEach(el=>{
    const key = el.dataset.i18n
    if(dict[key]){
        el.innerText = dict[key]
    }
})

}

function t(key, vars = {}){
let str = dict[key] || key

Object.keys(vars).forEach(k=>{
    str = str.replace(`{${k}}`, vars[k])
})

return str
}

function changeLang(lang){
loadLang(lang)
}

/* ===== UI ===== */

function toggleLangMenu(){

const menu = document.getElementById("langDropdown")

const visible = window.getComputedStyle(menu).display

menu.style.display = visible === "none" ? "block" : "none"
}

document.addEventListener("click", function(e){

const menu = document.getElementById("langDropdown")

if(!e.target.closest(".langMenu")){
menu.style.display = "none"
}

})
